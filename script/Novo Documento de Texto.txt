CREATE TABLE Soliciacao (
	id INT auto_increment not NULL,
	cliente VARCHAR(100),
	tempo VARCHAR(100),
	valor VARCHAR(18),
	dt_solictacao DATE,
	responsavel VARCHAR(500),
	status ENUM('Pendente', 'Aceito', 'Recusado'),
	PRIMARY KEY (id)	
);
